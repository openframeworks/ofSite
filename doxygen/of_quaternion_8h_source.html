454 ofVec3f ofQuaternion::operator*(const ofVec3f& v) const {
455  // nVidia SDK implementation
456  ofVec3f uv, uuv;
457  ofVec3f qvec(_v.x, _v.y, _v.z);
458  uv = qvec.getCrossed(v); //uv = qvec ^ v;
459  uuv = qvec.getCrossed(uv); //uuv = qvec ^ uv;
460  uv *= (2.0f * _v.w);
461  uuv *= 2.0f;
462  return v + uv + uuv;
463 }
